10macro_rules! cfe_time_type {
11    ($name:ident : $type_docstring:literal, $accessor_docstring:literal, $osal:ty) => {
12        #[doc = $type_docstring]
13        ///
14        /// Wraps `CFE_TIME_SysTime_t`.
15        #[doc(alias = "CFE_TIME_SysTime_t")]
16        #[derive(Clone, Copy, Debug)]
17        pub struct $name {
18            pub(crate) tm: CFE_TIME_SysTime_t,
19        }
20
21        impl $name {
22            #[doc = concat!("Creates a new `", stringify!($name), "` with the specified seconds/subseconds count.")]
23            #[inline]
24            pub const fn new(seconds: u32, subseconds: u32) -> Self {
25                Self { tm: CFE_TIME_SysTime_t { Seconds: seconds, Subseconds: subseconds } }
26            }
27
28            #[doc = concat!("Returns the number of whole seconds ", $accessor_docstring, ".")]
29            #[inline]
30            pub const fn seconds(self) -> u32 {
31                self.tm.Seconds
32            }
33
34            #[doc = concat!("Returns the fractional number of seconds ", $accessor_docstring)]
35            /// (in type-native units of 2<sup>&#8722;32</sup>&nbsp;seconds).
36            #[inline]
37            pub const fn subseconds(self) -> u32 {
38                self.tm.Subseconds
39            }
40
41            #[doc = concat!("Returns the fractional number of seconds ", $accessor_docstring)]
42            /// (in units of microseconds).
43            ///
44            /// Wraps `CFE_TIME_Sub2MicroSecs`.
45            #[doc(alias = "CFE_TIME_Sub2MicroSecs")]
46            #[inline]
47            pub fn microseconds(self) -> u32 {
48                unsafe { CFE_TIME_Sub2MicroSecs(self.tm.Subseconds) }
49            }
50        }
51
52        /// Wraps `CFE_TIME_Compare`.
53        impl PartialEq for $name {
54            #[doc(alias = "CFE_TIME_Compare")]
55            #[inline]
56            fn eq(&self, other: &Self) -> bool {
57                crate::sys::CFE_TIME_Compare_CFE_TIME_EQUAL == unsafe { CFE_TIME_Compare(self.tm, other.tm) }
58            }
59        }
60
61        impl Eq for $name {}
62
63        /// Wraps `CFE_TIME_Compare`.
64        impl Ord for $name {
65            #[doc(alias = "CFE_TIME_Compare")]
66            #[inline]
67            fn cmp(&self, other: &Self) -> Ordering {
68                match unsafe { CFE_TIME_Compare(self.tm, other.tm) } {
69                    crate::sys::CFE_TIME_Compare_CFE_TIME_A_LT_B => Ordering::Less,
70                    crate::sys::CFE_TIME_Compare_CFE_TIME_EQUAL => Ordering::Equal,
71                    _ => Ordering::Greater,
72                }
73            }
74        }
75
76        /// Wraps `CFE_TIME_Compare`.
77        impl PartialOrd for $name {
78            #[doc(alias = "CFE_TIME_Compare")]
79            #[inline]
80            fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
81                Some(self.cmp(other))
82            }
83        }
84
85        /// Wraps `OS_TimeGetTotalSeconds`, `OS_TimeGetMicrosecondsPart`, and `CFE_TIME_Micro2SubSecs`.
86        impl TryFrom<$osal> for $name {
87            type Error = core::num::TryFromIntError;
88
89            #[inline]
90            fn try_from(value: $osal) -> Result<Self, Self::Error> {
91                let seconds: u32 = value.total_seconds().try_into()?;
92                let subseconds = unsafe { CFE_TIME_Micro2SubSecs(value.microseconds_part()) };
93                Ok(Self::new(seconds, subseconds))
94            }
95        }
96
97        /// Wraps `OS_TimeAssembleFromMicroseconds` and `CFE_TIME_Sub2MicroSecs`.
98        impl From<$name> for $osal {
99            #[inline]
100            fn from(value: $name) -> Self {
101                let microseconds = unsafe { CFE_TIME_Sub2MicroSecs(value.subseconds()) };
102                <$osal>::from_microseconds(value.seconds() as i64, microseconds)
103            }
104        }
105    };
106}

119macro_rules! cfe_time_op {
120    ($trait:ident $method:ident $wrapped:ident $wrapped_str:literal : $($lhs:ty , $rhs:ty => $output:ty),*) => {
121        $(
122            #[doc = concat!("Wraps `", stringify!($wrapped), "`.")]
123            impl $trait<$rhs> for $lhs {
124                type Output = $output;
125
126                #[doc(alias = $wrapped_str)]
127                #[inline]
128                fn $method(self, rhs: $rhs) -> $output {
129                    Self::Output { tm: unsafe { $wrapped(self.tm, rhs.tm) } }
130                }
131            }
132        )*
133    };
134}

152/// Converts `microseconds` &mu;s to units of cFE sub-seconds (2<sup>&#8722;32</sup>&nbsp;seconds),
153/// or returns `!0` if `microseconds` is over `999_999`.
154///
155/// Wraps `CFE_TIME_Micro2SubSecs`.
156#[doc(alias = "CFE_TIME_Micro2SubSecs")]
157#[inline]
158pub fn micro_to_subsecs(microseconds: u32) -> u32 {
159    unsafe { CFE_TIME_Micro2SubSecs(microseconds) }
160}
161
162/// Converts `subseconds` cFE sub-seconds (2<sup>&#8722;32</sup>&nbsp;seconds) to microseconds.
163///
164/// Wraps `CFE_TIME_Sub2MicroSecs`.
165#[doc(alias = "CFE_TIME_Sub2MicroSecs")]
166#[inline]
167pub fn sub_to_microsecs(subseconds: u32) -> u32 {
168    unsafe { CFE_TIME_Sub2MicroSecs(subseconds) }
169}

171/// Returns the current spacecraft time,
172/// using the epoch specified in the mission configuration.
173///
174/// The time is returned either as seconds TAI or seconds UTC,
175/// depending on mission configuration.
176///
177/// Wraps `CFE_TIME_GetTime`.
178#[doc(alias = "CFE_TIME_GetTime")]
179#[inline]
180pub fn get_time() -> SysTime {
181    let tm = unsafe { CFE_TIME_GetTime() };
182    SysTime { tm }
