12/// A handle for a binary semaphore.
13///
14/// Wraps `osal_id_t`.
15#[doc(alias = "osal_id_t")]
16#[derive(Clone, Debug)]
17pub struct BinSem {
18    pub(crate) id: osal_id_t,
19}
20
21impl BinSem {
22    /// Attempts to create a new binary semaphore with name `name`,
23    /// initial value `initial_value`, and default options; if successful, returns it.
24    ///
25    /// Wraps `OS_BinSemCreate`.
26    #[doc(alias = "OS_BinSemCreate")]
27    #[inline]
28    pub fn new<S: AsRef<CStr> + ?Sized>(
29        name: &S,
30        initial_value: BinSemState,
31    ) -> Result<Self, OsalError> {
32        let mut id: osal_id_t = X_OS_OBJECT_ID_UNDEFINED;
33
34        unsafe { OS_BinSemCreate(&mut id, name.as_ref().as_ptr(), initial_value as u32, 0) }
35            .as_osal_status()?;
36
37        if id != X_OS_OBJECT_ID_UNDEFINED {
38            Ok(Self { id })
39        } else {
40            Err(OsalError::OS_ERR_INVALID_ID)
41        }
42    }
43
44    /// If a binary semaphore with the name `name` exists, returns `Ok(Some(`a handle to it`))`.
45    ///
46    /// If no binary semaphore with the name exists, returns `Ok(None)`.
47    /// If an error occurred, returns `Err(err_code)`.
48    ///
49    /// Wraps `OS_BinSemGetIdByName`.
50    #[doc(alias = "OS_BinSemGetIdByName")]
51    #[inline]
52    pub fn find_by_name<S: AsRef<CStr> + ?Sized>(name: &S) -> Result<Option<Self>, OsalError> {
53        let mut id: osal_id_t = X_OS_OBJECT_ID_UNDEFINED;
54
55        match unsafe { OS_BinSemGetIdByName(&mut id, name.as_ref().as_ptr()) }.as_osal_status() {
56            Ok(_) => {
57                if id != X_OS_OBJECT_ID_UNDEFINED {
58                    Ok(Some(Self { id }))
59                } else {
60                    Err(OsalError::OS_ERR_INVALID_ID)
61                }
62            }
63            Err(OsalError::OS_ERR_NAME_NOT_FOUND) => Ok(None),
64            Err(err) => Err(err),
65        }
66    }
67
68    /// Decrements the semaphore value, blocking until it is non-zero if needed.
69    ///
70    /// Wraps `OS_BinSemTake`.
71    #[doc(alias = "OS_BinSemTake")]
72    #[inline]
73    pub fn take(&self) -> Result<(), OsalError> {
74        unsafe { OS_BinSemTake(self.id) }.as_osal_status()?;
75
76        Ok(())
77    }
78
79    /// Decrements the semaphore value, blocking for up to `timeout_ms` milliseconds if need be.
80    ///
81    /// Returns `Ok(true)` if a lock was obtained before timing out,
82    /// `Ok(false)` if the request timed out,
83    /// or `Err(err_code)` if an error occurred.
84    ///
85    /// Wraps `OS_BinSemTimedWait`.
86    #[doc(alias = "OS_BinSemTimedWait")]
87    #[inline]
88    pub fn timed_wait(&self, timeout_ms: u32) -> Result<bool, OsalError> {
89        match unsafe { OS_BinSemTimedWait(self.id, timeout_ms) }.as_osal_status() {
90            Ok(_) => Ok(true),
91            Err(OsalError::OS_SEM_TIMEOUT) => Ok(false),
92            Err(err) => Err(err),
93        }
94    }
95
96    /// Increments the semaphore value, waking up a blocked thread (if any).
97    ///
98    /// Wraps `OS_BinSemGive`.
99    #[doc(alias = "OS_BinSemGive")]
100    #[inline]
101    pub fn give(&self) -> Result<(), OsalError> {
102        unsafe { OS_BinSemGive(self.id) }.as_osal_status()?;
103
104        Ok(())
105    }
106
107    /// Unblocks all tasks blocking on the semaphore without incrementing or decrementing its value.
108    ///
109    /// Wraps `OS_BinSemFlush`.
110    #[doc(alias = "OS_BinSemFlush")]
111    #[inline]
112    pub fn flush(&self) -> Result<(), OsalError> {
113        unsafe { OS_BinSemFlush(self.id) }.as_osal_status()?;
114
115        Ok(())
116    }
117
118    /// Deletes the binary semaphore.
119    ///
120    /// Wraps `OS_BinSemDelete`.
121    #[doc(alias = "OS_BinSemDelete")]
122    #[inline]
123    pub fn delete(self) -> Result<(), OsalError> {
124        unsafe { OS_BinSemDelete(self.id) }.as_osal_status()?;
125
126        Ok(())
127    }
128
129    /// If successful, returns details about the binary semaphore.
130    ///
131    /// Wraps `OS_BinSemGetInfo`.
132    #[doc(alias = "OS_BinSemGetInfo")]
133    #[inline]
134    pub fn info(&self) -> Result<BinSemProperties, OsalError> {
135        let mut props = OS_bin_sem_prop_t {
136            name:    [b'\0' as c_char; MAX_NAME_LEN],
137            creator: X_OS_OBJECT_ID_UNDEFINED,
138            value:   0,
139        };
140
141        unsafe { OS_BinSemGetInfo(self.id, &mut props) }.as_osal_status()?;
142
143        Ok(BinSemProperties {
144            name:    CStrBuf::new(&props.name),
145            creator: ObjectId { id: props.creator },
146            value:   props.value,
147        })
148    }
149
150    /// Returns the [`ObjectId`] for the mutex.
151    #[inline]
152    pub fn as_id(&self) -> ObjectId {
153        ObjectId { id: self.id }
154    }
155}

369impl MutSem {
370    /// Tries to create a new mutex semaphore with default options;
371    /// if successful, returns it.
372    ///
373    /// Per the cFE Users Guide, mutexes are always created in the unlocked state.
374    ///
375    /// Wraps `OS_MutSemCreate`.
376    #[doc(alias = "OS_MutSemCreate")]

379        let mut id: osal_id_t = X_OS_OBJECT_ID_UNDEFINED;
380
381        unsafe { OS_MutSemCreate(&mut id, sem_name.as_ref().as_ptr(), 0) }.as_osal_status()?;
382
383        if id != X_OS_OBJECT_ID_UNDEFINED {
384            Ok(Self { id })
385        } else {
386            Err(OsalError::OS_ERR_INVALID_ID)
387        }
388    }
389
390    /// If a mutex with the name `name` exists, returns `Ok(Some(`a handle to it`)`.
391    ///
392    /// If no mutex with the name exists, returns `Ok(None)`.
393    /// If an error occurred, returns `Err(err_code)`.
394    ///
395    /// Wraps `OS_MutSemGetIdByName`.
396    #[doc(alias = "OS_MutSemGetIdByName")]
397    #[inline]
398    pub fn find_by_name<S: AsRef<CStr> + ?Sized>(name: &S) -> Result<Option<Self>, OsalError> {
399        let mut id: osal_id_t = X_OS_OBJECT_ID_UNDEFINED;
400
401        match unsafe { OS_MutSemGetIdByName(&mut id, name.as_ref().as_ptr()) }.as_osal_status() {
402            Ok(_) => {
403                if id != X_OS_OBJECT_ID_UNDEFINED {
404                    Ok(Some(Self { id }))
405                } else {
406                    Err(OsalError::OS_ERR_INVALID_ID)
407                }
408            }
409            Err(OsalError::OS_ERR_NAME_NOT_FOUND) => Ok(None),
410            Err(err) => Err(err),
411        }
412    }
413
414    /// Attempts to acquire the mutex, blocking until it does.
415    /// Assuming nothing went wrong acquiring, runs the closure, then releases the mutex.
416    ///
417    /// Wraps `OS_MutSemTake` and `OS_MutSemGive`.
418    #[doc(alias("OS_MutSemTake", "OS_MutSemGive"))]
419    #[inline]
420    pub fn lock<T, F: FnOnce() -> T>(&self, closure: F) -> Result<T, OsalError> {
421        self.take()?;
422
423        struct MutGuard {
424            x: MutSem,
425        }
426        impl Drop for MutGuard {
427            fn drop(&mut self) {
428                let _ = self.x.give();
429            }
430        }
431
432        let guard = MutGuard { x: self.clone() };
433
434        let val = closure();
435
436        drop(guard);
437        Ok(val)
438    }
439
440    // TODO: determine if this should be `pub`
441    /// If successful, acquires the mutex; if the mutex is currently acquired, this thread will block until it does acquire it.
442    ///
443    /// Wraps `OS_MutSemTake`.
444    #[doc(alias = "OS_MutSemTake")]
445    #[inline]
446    fn take(&self) -> Result<(), OsalError> {
447        unsafe { OS_MutSemTake(self.id) }.as_osal_status()?;
448
449        Ok(())
450    }
451
452    // TODO: determine if this should be `pub`

458    fn give(&self) -> Result<(), OsalError> {

526macro_rules! owned_sem_variant {
527    ($type_name:ident, $wrapped_type:ty, $destructor:ident, $constructor:ident $(; $cparam:ident: $ctype:ty)*) => {
528        #[doc = concat!("A wrapper around a [`", stringify!($wrapped_type), "`] that automatically deletes it when dropped.")]
529        pub struct $type_name {
530            sem: $wrapped_type,
531        }
532
533        impl $type_name {
534            #[doc = concat!("Like [`", stringify!($wrapped_type), "::new`], but creates an owned semaphore instead.")]
535            #[doc = "\n\n"]
536            #[doc = concat!("Wraps `", stringify!($constructor), "`.")]
537            #[inline]
538            pub fn new<S: AsRef<CStr> + ?Sized>(sem_name: &S $(, $cparam: $ctype )*) -> Result<Self, OsalError> {
539                <$wrapped_type>::new(sem_name $(, $cparam)*).map(|sem| $type_name { sem })
540            }
541        }
542
543        impl core::ops::Deref for $type_name {
544            type Target = $wrapped_type;
545
546            #[inline]
547            fn deref(&self) -> &$wrapped_type {
548                &self.sem
549            }
550        }
551
552        #[doc = concat!("Wraps `", stringify!($destructor), "`.")]
553        impl Drop for $type_name {
554            #[inline]
555            fn drop(&mut self) {
556                let _ = unsafe { $destructor(self.sem.id) };
557            }
558        }
559
560        #[doc = concat!("Takes the wrapped [`", stringify!($wrapped_type), "`] out of the [`", stringify!($type_name), "`] wrapper without dropping it.")]
561        impl From<$type_name> for $wrapped_type {
562            #[inline]
563            fn from(o_sem: $type_name) -> Self {
564                let x = Self { id: o_sem.sem.id };
565                let _ = core::mem::ManuallyDrop::new(o_sem);
566                x
567            }
568        }
569    };
570}
